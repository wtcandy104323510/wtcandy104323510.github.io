OpenCV 人臉偵測學習


以《維京傳奇》劇照為例

以 Anaconda 中的 Spyder 為編譯器來進行練習

安裝 OpenCV 函式庫

pip install opencv-python

下載 OpenCV 官方的人臉特徵模型

https://github.com/opencv/opencv/tree/4.x/data

人臉辨識程式


# 讀取圖片   
img = cv2.imread('1005.jpg')

# 將圖片轉為灰階
gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)

# 讀取opencv的特徵模型xml檔
pictPath = r'E:\Wistron\Python\opencv\OpenCVDatabase\opencv\sources\data\haarcascades\haarcascade_frontalface_default.xml'


# 利用 opencv 的分類器建立辨識檔案物件
faceCascade = cv2.CascadeClassifier(pictPath)

# 偵測人臉
faceRect = faceCascade.detectMultiScale(gray, 1.1, 5)    

print(len(faceRect))

# 利用 for 迴圈將偵測到的人臉繪製方框
for (x, y, w, h) in faceRect:
    cv2.rectangle(img, (x,y), (x+w,y+h), (0, 255, 0), 2)

# 顯示繪製完的圖片
cv2.imshow('img', img)

cv2.waitKey(0)

cv2.destroyAllWindows()

detectMultiScale() 函式介紹

faceRect = faceCascade.detectMultiScale(img, scaleFactor, minNeighbors)

# img：圖片來源（建議使用灰階影像）
# scaleFactor：前後圖片偵測的比例係數，係數愈大，縮小比例就愈快
# minNeighbors：圖片被偵測到的次數依據，數字愈大，代表偵測條件愈嚴謹
